```
// ==UserScript==
// @name         超星学习通-视频自动完成器
// @namespace    http://tampermonkey.net/
// @version      3.3
// @description  在iframe中检测未完成视频并自动播放，完成后自动返回
// @author       You
// @match        https://task.chaoxing.com/*
// @match        https://mooc1.chaoxing.com/*
// @match        https://mooc2.chaoxing.com/*
// @match        https://*.chaoxing.com/*
// @grant        GM_notification
// @run-at       document-end
// ==/UserScript==

(function() {
    'use strict';

    console.log('🎥 超星视频自动完成器 已启动');

    // 状态管理
    let isRunning = false;
    let videoQueue = [];
    let currentVideoIndex = 0;
    let checkInterval = null;
    let isInVideoPage = false;
    let targetIframe = null;
    let currentVideoElement = null;
    let lastProgressCheck = 0;
    let noProgressCount = 0;

    // 系统通知
    function showNotification(title, message) {
        console.log(`✅ ${title}: ${message}`);
        try {
            if (typeof GM_notification !== 'undefined') {
                GM_notification({ title: title, text: message, timeout: 5000 });
            } else if (window.Notification && Notification.permission === 'granted') {
                new Notification(title, { body: message });
            } else {
                alert(`${title}\n${message}`);
            }
        } catch(e) {}
    }

    function requestNotificationPermission() {
        if (window.Notification && Notification.permission === 'default') {
            Notification.requestPermission();
        }
    }

    function getTargetIframe() {
        if (targetIframe) return targetIframe;
        const iframe = document.getElementById('child');
        if (iframe) {
            console.log('找到目标iframe');
            targetIframe = iframe;
            return iframe;
        }
        return null;
    }

    // 在iframe中查找视频任务
    function scanVideosInIframe() {
        const iframe = getTargetIframe();
        if (!iframe) return [];

        try {
            const iframeDoc = iframe.contentDocument || iframe.contentWindow.document;
            if (!iframeDoc) return [];

            console.log('开始在iframe中扫描视频...');

            const tasks = [];
            const taskItems = iframeDoc.querySelectorAll('.xtask_item');
            console.log(`在iframe中找到 ${taskItems.length} 个任务项`);

            taskItems.forEach((item) => {
                const text = item.textContent?.trim();
                const html = item.outerHTML.toLowerCase();

                let isVideo = html.includes('视频') || html.includes('video') ||
                              text?.includes('视频');

                if (isVideo) {
                    let isCompleted = false;
                    if (html.includes('已完成') || html.includes('green') ||
                        item.querySelector('.green') || item.querySelector('.colorGreen')) {
                        isCompleted = true;
                    }
                    if (text?.includes('未完成') || item.querySelector('.colorRed')) {
                        isCompleted = false;
                    }

                    const title = item.querySelector('.xname')?.textContent?.trim() ||
                                 text?.substring(0, 50) || '未知视频';
                    tasks.push({
                        element: item,
                        text: title,
                        completed: isCompleted
                    });
                    console.log(`${isCompleted ? '✅' : '❌'} ${title}`);
                }
            });

            return tasks;
        } catch(e) {
            console.log('无法访问iframe内容:', e.message);
            return [];
        }
    }

    // 创建控制面板
    function createPanel() {
        const oldPanel = document.getElementById('videoAutoPanel');
        if (oldPanel) oldPanel.remove();

        const panel = document.createElement('div');
        panel.id = 'videoAutoPanel';
        panel.style.cssText = `
            position: fixed;
            top: 50px;
            right: 10px;
            z-index: 999999;
            background: white;
            border: 3px solid #27ae60;
            border-radius: 10px;
            padding: 15px;
            width: 320px;
            box-shadow: 0 4px 20px rgba(0,0,0,0.3);
            font-family: Arial, sans-serif;
        `;

        panel.innerHTML = `
            <h3 style="margin:0 0 10px 0; color:#27ae60;">🎬 视频自动完成器</h3>
            <div style="background: #f0f0f0; padding: 8px; border-radius: 5px; margin-bottom: 10px;">
                <div><strong>未完成视频:</strong> <span id="remainingTasks" style="color:#ff4757;">0</span></div>
                <div><strong>当前播放:</strong> <span id="currentVideo">无</span></div>
                <div><strong>播放进度:</strong> <span id="playProgress">0%</span></div>
                <div><strong>状态:</strong> <span id="runStatus">空闲</span></div>
            </div>
            <div id="videoList" style="margin-bottom: 10px; max-height: 180px; overflow-y: auto; font-size:11px;"></div>
            <div style="display: flex; gap: 5px;">
                <button id="startAuto" style="flex:2; background:#27ae60; color:white; border:none; padding:6px; border-radius:4px; cursor:pointer;">▶ 一键开始</button>
                <button id="stopAuto" style="flex:1; background:#e74c3c; color:white; border:none; padding:6px; border-radius:4px; cursor:pointer;">⏹ 停止</button>
                <button id="scanAgain" style="flex:1; background:#3498db; color:white; border:none; padding:6px; border-radius:4px; cursor:pointer;">🔄 扫描</button>
            </div>
            <div id="logPanel" style="margin-top:8px; font-size:10px; color:#666; border-top:1px solid #ddd; padding-top:5px; max-height:60px; overflow-y:auto;"></div>
        `;

        document.body.appendChild(panel);

        document.getElementById('scanAgain').onclick = () => scanAndDisplay();
        document.getElementById('startAuto').onclick = () => startAutoPlay();
        document.getElementById('stopAuto').onclick = () => stopAutoPlay();

        makeDraggable(panel);
    }

    function makeDraggable(element) {
        let isDragging = false;
        let offsetX, offsetY;

        element.addEventListener('mousedown', (e) => {
            if (e.target.tagName === 'BUTTON') return;
            isDragging = true;
            offsetX = e.clientX - element.offsetLeft;
            offsetY = e.clientY - element.offsetTop;
        });

        document.addEventListener('mousemove', (e) => {
            if (isDragging) {
                element.style.left = (e.clientX - offsetX) + 'px';
                element.style.top = (e.clientY - offsetY) + 'px';
                element.style.right = 'auto';
            }
        });

        document.addEventListener('mouseup', () => {
            isDragging = false;
        });
    }

    function addLog(message) {
        console.log(message);
        const logPanel = document.getElementById('logPanel');
        if (logPanel) {
            const entry = document.createElement('div');
            entry.textContent = `[${new Date().toLocaleTimeString().slice(0,5)}] ${message}`;
            logPanel.appendChild(entry);
            while (logPanel.children.length > 8) logPanel.removeChild(logPanel.children[0]);
            logPanel.scrollTop = logPanel.scrollHeight;
        }
    }

    function updateStatus(status) {
        const span = document.getElementById('runStatus');
        if (span) {
            span.textContent = status;
            span.style.color = status === '运行中' ? '#27ae60' : (status === '已完成' ? '#27ae60' : '#666');
        }
    }

    function scanAndDisplay() {
        addLog('🔍 扫描视频...');
        const tasks = scanVideosInIframe();
        const incomplete = tasks.filter(t => !t.completed);
        videoQueue = incomplete;

        document.getElementById('remainingTasks').textContent = incomplete.length;

        const listDiv = document.getElementById('videoList');
        if (incomplete.length === 0) {
            listDiv.innerHTML = '<div style="color:#666; text-align:center;">✅ 没有未完成的视频</div>';
        } else {
            let html = '<div><strong>📋 待播放列表:</strong></div>';
            incomplete.forEach((v, i) => {
                html += `<div style="margin:4px 0; padding:4px; background:#f8f9fa;">${i+1}. ${v.text.substring(0, 35)}</div>`;
            });
            listDiv.innerHTML = html;
        }
        addLog(`找到 ${tasks.length} 个视频，未完成 ${incomplete.length} 个`);
    }

    function startAutoPlay() {
        if (isRunning) {
            addLog('已在运行中');
            return;
        }

        addLog('🚀 开始自动播放');
        updateStatus('运行中');

        scanAndDisplay();

        if (videoQueue.length === 0) {
            addLog('没有未完成的视频');
            updateStatus('空闲');
            showNotification('视频完成', '所有视频已完成！');
            return;
        }

        isRunning = true;
        currentVideoIndex = 0;
        addLog(`共 ${videoQueue.length} 个视频待播放`);

        setTimeout(() => playCurrentVideo(), 1000);
    }

    function stopAutoPlay() {
        isRunning = false;
        isInVideoPage = false;
        if (checkInterval) clearInterval(checkInterval);
        document.getElementById('currentVideo').textContent = '无';
        document.getElementById('playProgress').textContent = '0%';
        updateStatus('停止');
        addLog('⏹️ 已停止');
    }

    function playCurrentVideo() {
        if (!isRunning) return;

        if (currentVideoIndex >= videoQueue.length) {
            addLog('🎉 所有视频已完成！');
            updateStatus('已完成');
            showNotification('恭喜！', '所有视频已完成！');
            stopAutoPlay();
            return;
        }

        const video = videoQueue[currentVideoIndex];
        if (!video || !video.element) {
            addLog('视频元素不存在，跳过');
            currentVideoIndex++;
            playCurrentVideo();
            return;
        }

        const title = video.text.substring(0, 30);
        document.getElementById('currentVideo').textContent = title;
        addLog(`▶️ 播放 ${currentVideoIndex + 1}/${videoQueue.length}: ${title}`);

        try {
            video.element.style.border = '2px solid #27ae60';
            video.element.style.backgroundColor = 'rgba(39, 174, 96, 0.1)';
            video.element.scrollIntoView({ behavior: 'smooth', block: 'center' });
            video.element.click();

            isInVideoPage = true;
            currentVideoElement = null;
            lastProgressCheck = 0;
            noProgressCount = 0;

            setTimeout(() => handleVideoPage(), 4000);
        } catch(e) {
            addLog(`点击失败: ${e.message}`);
            currentVideoIndex++;
            playCurrentVideo();
        }
    }

    // 查找视频元素（包括iframe）
    function findVideoElement() {
        let video = document.querySelector('video');
        if (video) return video;

        const iframes = document.querySelectorAll('iframe');
        for (let iframe of iframes) {
            try {
                const doc = iframe.contentDocument || iframe.contentWindow.document;
                if (doc) {
                    video = doc.querySelector('video');
                    if (video) return video;
                }
            } catch(e) {}
        }
        return null;
    }

    // 核心：通过监控视频播放进度判断是否完成
    function handleVideoPage() {
        addLog('处理视频页面');

        // 查找视频元素
        currentVideoElement = findVideoElement();

        if (!currentVideoElement) {
            addLog('未找到视频，等待重试...');
            setTimeout(() => handleVideoPage(), 2000);
            return;
        }

        addLog('找到视频，开始播放');

        // 播放视频
        currentVideoElement.play().catch(e => {
            addLog(`自动播放失败: ${e.message}`);
            currentVideoElement.muted = true;
            currentVideoElement.play();
        });

        currentVideoElement.playbackRate = 1.5;
        addLog(`播放速度: ${currentVideoElement.playbackRate}x`);

        // 清除旧监控
        if (checkInterval) clearInterval(checkInterval);

        // 启动监控 - 核心逻辑
        checkInterval = setInterval(() => {
            if (!isRunning || !isInVideoPage) {
                clearInterval(checkInterval);
                return;
            }

            // 重新获取视频元素（防止元素变化）
            const video = findVideoElement();
            if (!video) return;

            const duration = video.duration;
            const currentTime = video.currentTime;

            // 如果视频时长无效，继续等待
            if (!duration || isNaN(duration) || duration === 0) {
                addLog('等待视频加载...');
                return;
            }

            const percentage = (currentTime / duration * 100).toFixed(1);
            document.getElementById('playProgress').textContent = `${percentage}% (${Math.floor(currentTime)}/${Math.floor(duration)}秒)`;

            // 关键判断：是否播放完成
            // 1. 当前播放时间 >= 总时长 - 0.5秒（播放到结尾）
            // 2. 或者播放时间已经达到总时长（四舍五入）
            const isFinished = currentTime >= duration - 0.5 || Math.floor(currentTime) >= Math.floor(duration);

            if (isFinished) {
                addLog(`✅ 视频播放完成！总时长: ${Math.floor(duration)}秒，实际播放: ${Math.floor(currentTime)}秒`);
                clearInterval(checkInterval);
                closeVideoAndNext();
                return;
            }

            // 检查播放进度是否卡住（5秒无变化）
            if (currentTime === lastProgressCheck) {
                noProgressCount++;
                if (noProgressCount >= 5) { // 5秒没动
                    addLog(`⚠️ 视频卡在 ${Math.floor(currentTime)}/${Math.floor(duration)}秒，尝试重新播放`);
                    video.play().catch(e => {});
                    noProgressCount = 0;
                }
            } else {
                noProgressCount = 0;
                lastProgressCheck = currentTime;
            }

            // 每10秒输出一次进度
            if (Math.floor(currentTime) % 10 === 0 && Math.floor(currentTime) !== lastProgressCheck) {
                addLog(`播放中: ${percentage}% (${Math.floor(currentTime)}/${Math.floor(duration)}秒)`);
            }
        }, 1000);
    }

    function closeVideoAndNext() {
        addLog('关闭视频页面');
        isInVideoPage = false;
        currentVideoIndex++;

        // 尝试关闭
        try {
            const closeBtns = document.querySelectorAll('.close, .icon-close, [class*="close"]');
            for (let btn of closeBtns) {
                if (btn.offsetParent !== null) {
                    btn.click();
                    break;
                }
            }
            const backBtns = document.querySelectorAll('.back, .return, .goback');
            for (let btn of backBtns) {
                if (btn.offsetParent !== null) {
                    btn.click();
                    break;
                }
            }
        } catch(e) {}

        history.back();

        setTimeout(() => {
            if (isRunning) {
                scanAndDisplay();
                playCurrentVideo();
            }
        }, 3000);
    }

    function waitForIframe() {
        const iframe = getTargetIframe();
        if (iframe) {
            if (iframe.contentDocument && iframe.contentDocument.readyState === 'complete') {
                createPanel();
                scanAndDisplay();
                addLog('✅ 脚本加载完成，点击"一键开始"');
            } else {
                iframe.addEventListener('load', () => {
                    createPanel();
                    scanAndDisplay();
                    addLog('✅ 脚本加载完成，点击"一键开始"');
                });
            }
        } else {
            setTimeout(waitForIframe, 1000);
        }
    }

    requestNotificationPermission();

    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', waitForIframe);
    } else {
        waitForIframe();
    }

})();
```
