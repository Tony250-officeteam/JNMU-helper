// ==UserScript==
// @name         超星讨论区自动复制+F2一键粘贴（增强检测版）
// @namespace    http://tampermonkey.net/
// @version       5.0
// @description  进入页面自动复制老师问题并添加指令，带自我检测和重试机制，按F2自动滚动到底部并粘贴
// @author       YourName
// @match        *://*.chaoxing.com/course/topic/v3/bbs/*
// @match        https://groupweb.chaoxing.com/course/topicDiscuss/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // 全局状态变量
    let copyStatus = {
        success: false,
        text: '',
        retryCount: 0,
        maxRetries: 3
    };

    // ---------- 1. 增强版：精确查找老师的问题 ----------
    function findTeacherQuestion() {
        // 策略1: 查找讨论区主要内容区域（包含问题文本）
        const possibleContainers = [
            // 直接匹配您提供的结构
            '.richText.richtext',
            '.richText',
            '.topic-content',
            '.bbs-topic-item .content',
            '.contentHtml',
            '.question-content',
            '.description',
            // 讨论区常见容器
            'div[class*="topic"]',
            'div[class*="question"]',
            'div[class*="content"]'
        ];

        let foundText = null;

        // 遍历可能的容器
        for (let sel of possibleContainers) {
            const elements = document.querySelectorAll(sel);
            for (let el of elements) {
                // 获取文本内容
                let text = el.innerText?.trim();
                if (text && text.length > 20) { // 问题通常较长
                    // 检查是否包含常见问题关键词
                    if (text.includes('？') || text.includes('?') ||
                        text.includes('思考题') || text.includes('简述') ||
                        text.includes('为什么') || text.includes('说明')) {
                        foundText = text;
                        break;
                    }
                    // 如果没有明显关键词但文本较长，也作为候选
                    if (!foundText && text.length > 50) {
                        foundText = text;
                    }
                }
            }
            if (foundText) break;
        }

        // 策略2: 查找所有可能的段落
        if (!foundText) {
            const paragraphs = document.querySelectorAll('p');
            for (let p of paragraphs) {
                let text = p.innerText?.trim();
                if (text && text.length > 30) {
                    // 检查是否包含问题特征
                    if (text.match(/[0-9]+\./) || text.includes('？') || text.includes('?') ||
                        text.includes('简述') || text.includes('说明') || text.includes('分析')) {
                        foundText = text;
                        break;
                    }
                }
            }
        }

        // 策略3: 获取整个讨论区域
        if (!foundText) {
            const mainContent = document.querySelector('.main-content, .bbs-container, .discuss-area');
            if (mainContent) {
                foundText = mainContent.innerText?.trim();
            }
        }

        // 清理和格式化文本
        if (foundText) {
            // 移除多余空白行
            foundText = foundText.replace(/\n\s*\n/g, '\n').trim();
            console.log('找到的问题文本:', foundText.substring(0, 200));
            return foundText;
        }

        return null;
    }

    // ---------- 2. 查找回复输入框 ----------
    function findReplyTextarea() {
        // 主要目标
        const textarea = document.querySelector('textarea[autoheight="true"][placeholder*="回复"]');
        if (textarea) return textarea;

        // 备选方案
        const backupSelectors = [
            'textarea[placeholder*="回复"]',
            'textarea[placeholder*="评论"]',
            'textarea[placeholder*="发表"]',
            '.replyEdit .textareawrap textarea',
            '.editContainer textarea',
            '.reply-input textarea',
            'div[contenteditable="true"]'
        ];

        for (let sel of backupSelectors) {
            const el = document.querySelector(sel);
            if (el && (el.tagName === 'TEXTAREA' || el.getAttribute('contenteditable') === 'true')) {
                return el;
            }
        }

        return null;
    }

    // ---------- 3. 带检测的复制函数 ----------
    async function copyWithVerification(text, retryCallback) {
        // 先尝试使用现代剪贴板API
        if (navigator.clipboard && navigator.clipboard.writeText) {
            try {
                await navigator.clipboard.writeText(text);

                // 自我检测：验证剪贴板内容是否正确
                const verified = await verifyClipboardContent(text);
                if (verified) {
                    console.log('✓ 复制成功且内容验证通过');
                    copyStatus.success = true;
                    copyStatus.text = text;
                    updateHintStatus('success');
                    return true;
                } else {
                    console.warn('⚠ 剪贴板内容验证失败，可能被其他程序修改');
                    return fallbackCopyWithVerification(text, retryCallback);
                }
            } catch (err) {
                console.error('剪贴板API写入失败', err);
                return fallbackCopyWithVerification(text, retryCallback);
            }
        } else {
            return fallbackCopyWithVerification(text, retryCallback);
        }
    }

    // 备用复制方法 + 验证
    function fallbackCopyWithVerification(text, retryCallback) {
        const textarea = document.createElement('textarea');
        textarea.value = text;
        document.body.appendChild(textarea);
        textarea.select();

        try {
            const success = document.execCommand('copy');
            document.body.removeChild(textarea);

            if (success) {
                setTimeout(async () => {
                    const verified = await verifyClipboardContent(text);
                    if (verified) {
                        copyStatus.success = true;
                        copyStatus.text = text;
                        updateHintStatus('success');
                    } else {
                        console.warn('备用复制后验证失败');
                        handleCopyFailure(text, retryCallback);
                    }
                }, 500);
                return true;
            } else {
                document.body.removeChild(textarea);
                handleCopyFailure(text, retryCallback);
                return false;
            }
        } catch (err) {
            document.body.removeChild(textarea);
            console.error('备用复制失败', err);
            handleCopyFailure(text, retryCallback);
            return false;
        }
    }

    // 验证剪贴板内容
    async function verifyClipboardContent(expectedText) {
        if (!navigator.clipboard || !navigator.clipboard.readText) {
            return true;
        }

        try {
            const clipboardText = await navigator.clipboard.readText();
            const normalizedExpected = expectedText.replace(/\s+/g, ' ').trim();
            const normalizedActual = clipboardText.replace(/\s+/g, ' ').trim();

            // 检查是否包含关键内容（前100个字符）
            const keyContent = expectedText.substring(0, 100);
            return normalizedActual.includes(keyContent) || normalizedExpected === normalizedActual;
        } catch (err) {
            console.log('无法读取剪贴板进行验证，可能是权限问题');
            return true;
        }
    }

    // 处理复制失败
    function handleCopyFailure(text, retryCallback) {
        copyStatus.retryCount++;

        if (copyStatus.retryCount <= copyStatus.maxRetries) {
            console.log(`⏳ 复制失败，第${copyStatus.retryCount}次重试...`);
            updateHintStatus('retrying', copyStatus.retryCount);

            setTimeout(() => {
                if (retryCallback) {
                    retryCallback();
                } else {
                    copyWithVerification(text, handleCopyFailure);
                }
            }, 1000 * copyStatus.retryCount);
        } else {
            console.error('❌ 复制失败，已达最大重试次数');
            copyStatus.success = false;
            updateHintStatus('failed');
        }
    }

    // ---------- 4. 自动复制主函数（增强版）----------
    function autoCopyTeacherQuestion() {
        console.log('开始查找问题...');
        const question = findTeacherQuestion();

        if (!question) {
            console.log('未找到老师的问题，5秒后重试...');
            updateHintStatus('searching');

            setTimeout(() => {
                const retryQuestion = findTeacherQuestion();
                if (retryQuestion) {
                    const finalText = retryQuestion + '\n\n简要回答，不要使用任何markdown的特殊记号';
                    copyWithVerification(finalText, () => autoCopyTeacherQuestion());
                } else {
                    console.log('最终仍未找到老师的问题');
                    updateHintStatus('not-found');
                }
            }, 5000);
            return;
        }

        const finalText = question + '\n\n简要回答，不要使用任何markdown的特殊记号';
        console.log('准备复制的问题长度:', finalText.length);
        copyWithVerification(finalText, () => autoCopyTeacherQuestion());
    }

    // ---------- 5. 粘贴操作 ----------
    function pasteToBottomInput() {
        if (!copyStatus.success) {
            alert('⚠ 剪贴板中可能没有正确的内容，建议先等待自动复制完成');
            autoCopyTeacherQuestion();
        }

        window.scrollTo(0, document.body.scrollHeight);

        setTimeout(() => {
            const inputElement = findReplyTextarea();

            if (!inputElement) {
                alert('未找到回复输入框，请手动粘贴');
                return;
            }

            if (navigator.clipboard && navigator.clipboard.readText) {
                navigator.clipboard.readText().then(text => {
                    if (inputElement.tagName === 'TEXTAREA' || inputElement.tagName === 'INPUT') {
                        inputElement.value = text;
                    } else if (inputElement.isContentEditable) {
                        inputElement.innerText = text;
                    } else {
                        inputElement.value = text;
                    }

                    inputElement.dispatchEvent(new Event('input', { bubbles: true }));
                    inputElement.dispatchEvent(new Event('change', { bubbles: true }));
                    inputElement.focus();

                    console.log('✓ 已粘贴到输入框');
                }).catch(err => {
                    console.error('读取剪贴板失败', err);
                    alert('无法读取剪贴板，请手动Ctrl+V粘贴');
                });
            } else {
                alert('浏览器不支持自动读取剪贴板，请手动Ctrl+V粘贴');
            }
        }, 300);
    }

    // ---------- 6. 监听F2 ----------
    document.addEventListener('keydown', function(e) {
        if (e.key === 'F2' || e.keyCode === 113) {
            e.preventDefault();
            console.log('F2按下，执行粘贴操作...');
            pasteToBottomInput();
        }
    });

    // ---------- 7. 增强的可视化提示 ----------
    let hintElement = null;

    function createHint() {
        if (hintElement) return hintElement;

        hintElement = document.createElement('div');
        hintElement.style.position = 'fixed';
        hintElement.style.bottom = '20px';
        hintElement.style.right = '20px';
        hintElement.style.zIndex = 9999;
        hintElement.style.backgroundColor = '#2196F3';
        hintElement.style.color = 'white';
        hintElement.style.padding = '12px 20px';
        hintElement.style.borderRadius = '25px';
        hintElement.style.boxShadow = '0 4px 15px rgba(0,0,0,0.3)';
        hintElement.style.fontSize = '14px';
        hintElement.style.cursor = 'pointer';
        hintElement.style.fontWeight = 'bold';
        hintElement.style.border = 'none';
        hintElement.style.display = 'flex';
        hintElement.style.alignItems = 'center';
        hintElement.style.gap = '10px';
        hintElement.style.transition = 'all 0.3s ease';

        const statusDot = document.createElement('span');
        statusDot.id = 'copy-status-dot';
        statusDot.style.display = 'inline-block';
        statusDot.style.width = '12px';
        statusDot.style.height = '12px';
        statusDot.style.borderRadius = '50%';
        statusDot.style.backgroundColor = '#FFA500';
        statusDot.style.animation = 'pulse 1.5s infinite';

        const style = document.createElement('style');
        style.textContent = `
            @keyframes pulse {
                0% { opacity: 1; transform: scale(1); }
                50% { opacity: 0.6; transform: scale(1.1); }
                100% { opacity: 1; transform: scale(1); }
            }
        `;
        document.head.appendChild(style);

        const textSpan = document.createElement('span');
        textSpan.id = 'copy-status-text';
        textSpan.textContent = '正在查找问题...';

        hintElement.appendChild(statusDot);
        hintElement.appendChild(textSpan);

        hintElement.onclick = function() {
            if (copyStatus.success) {
                pasteToBottomInput();
            } else {
                autoCopyTeacherQuestion();
            }
        };

        document.body.appendChild(hintElement);
        return hintElement;
    }

    // 更新提示状态
    function updateHintStatus(status, retryCount = 0) {
        const hint = createHint();
        const dot = document.getElementById('copy-status-dot');
        const text = document.getElementById('copy-status-text');

        if (!dot || !text) return;

        switch(status) {
            case 'success':
                dot.style.backgroundColor = '#4CAF50';
                dot.style.animation = 'none';
                text.textContent = '✅ 已自动复制问题，按F2粘贴';
                hint.style.backgroundColor = '#4CAF50';
                break;
            case 'searching':
                dot.style.backgroundColor = '#FFA500';
                dot.style.animation = 'pulse 1.5s infinite';
                text.textContent = '🔍 正在查找问题...';
                hint.style.backgroundColor = '#2196F3';
                break;
            case 'retrying':
                dot.style.backgroundColor = '#FFA500';
                dot.style.animation = 'pulse 1s infinite';
                text.textContent = `⏳ 复制失败，第${retryCount}次重试...`;
                hint.style.backgroundColor = '#FF9800';
                break;
            case 'failed':
                dot.style.backgroundColor = '#f44336';
                dot.style.animation = 'none';
                text.textContent = '❌ 复制失败，点击重试';
                hint.style.backgroundColor = '#f44336';
                break;
            case 'not-found':
                dot.style.backgroundColor = '#9e9e9e';
                dot.style.animation = 'none';
                text.textContent = '⚠ 未找到问题，请手动复制';
                hint.style.backgroundColor = '#9e9e9e';
                break;
        }
    }

    // ---------- 8. 初始化 ----------
    function init() {
        createHint();
        updateHintStatus('searching');

        // 延迟执行，确保页面完全加载
        setTimeout(() => {
            autoCopyTeacherQuestion();
        }, 1500);

        // 监听动态内容变化
        const observer = new MutationObserver(function(mutations) {
            if (!copyStatus.success) {
                autoCopyTeacherQuestion();
            }
        });

        observer.observe(document.body, {
            childList: true,
            subtree: true
        });

        // 定期检查剪贴板
        setInterval(async () => {
            if (copyStatus.success && copyStatus.text) {
                const verified = await verifyClipboardContent(copyStatus.text);
                if (!verified) {
                    console.log('检测到剪贴板内容被修改，重新复制...');
                    autoCopyTeacherQuestion();
                }
            }
        }, 30000);
    }

    // 启动
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', init);
    } else {
        init();
    }
})();
