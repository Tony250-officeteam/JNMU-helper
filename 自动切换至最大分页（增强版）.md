// ==UserScript==
// @name         自动切换至最大分页（增强版）
// @namespace    http://tampermonkey.net/
// @version      2.0
// @description  强制将 .pagination-page-list 切至最大页码并触发页面刷新
// @author       You
// @match        http://210.44.16.13/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const SELECTOR = 'select.pagination-page-list';

    function triggerEvents(element) {
        // 模拟鼠标操作序列
        const events = ['mousedown', 'mouseup', 'click', 'change', 'input'];
        events.forEach(eventType => {
            const event = new Event(eventType, { bubbles: true, cancelable: true });
            element.dispatchEvent(event);
        });

        // 如果页面有 jQuery，使用它来触发更完整的事件链
        if (typeof $ !== 'undefined' && $.fn && $.fn.jquery) {
            $(element).trigger('change');
            $(element).trigger('click');
        }

        // 部分 EasyUI 分页需要额外处理：尝试触发父组件的方法
        const paginationDiv = element.closest('.pagination');
        if (paginationDiv && typeof $ !== 'undefined' && $.fn.pagination) {
            try {
                // 尝试调用 EasyUI pagination 的 select 方法
                const pageNumber = parseInt(element.value, 10);
                $(paginationDiv).pagination('select', pageNumber);
                console.log(`[分页切换] 通过 EasyUI API 切换到第 ${pageNumber} 页`);
            } catch (e) {
                // 忽略错误，继续执行其他方式
            }
        }
    }

    function switchToLastPage(select) {
        const options = select.options;
        if (options.length === 0) return;

        const lastIndex = options.length - 1;
        if (select.selectedIndex === lastIndex) return;

        // 更改选中项
        select.selectedIndex = lastIndex;
        const lastValue = options[lastIndex].value;

        // 触发事件
        triggerEvents(select);

        console.log(`[分页切换] 已尝试切换到最大值：${lastValue}`);
    }

    function processSelect(select) {
        // 给元素添加短暂延迟，确保框架已完全初始化
        setTimeout(() => {
            switchToLastPage(select);
        }, 100);
    }

    // 处理已存在的元素
    function processExisting() {
        document.querySelectorAll(SELECTOR).forEach(processSelect);
    }

    // 监听动态添加的元素
    const observer = new MutationObserver(mutations => {
        for (const mutation of mutations) {
            for (const node of mutation.addedNodes) {
                if (node.nodeType !== Node.ELEMENT_NODE) continue;
                if (node.matches && node.matches(SELECTOR)) {
                    processSelect(node);
                }
                node.querySelectorAll?.(SELECTOR).forEach(processSelect);
            }
        }
    });

    observer.observe(document.body, { childList: true, subtree: true });

    // 页面加载完成后立即处理一次
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', processExisting);
    } else {
        processExisting();
    }
})();
